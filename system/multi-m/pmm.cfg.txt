;	PMM.COM、CS.COM、PMM.CFG はすべて同じﾃﾞｨﾚｸﾄﾘに置いてください。
;	<注意> PMM.COM は圧縮していないｵﾘｼﾞﾅﾙのものを使用してください。

;--- ﾊﾟﾗﾒｰﾀで使用できるマクロ文字 ------------- 書式例 ------
;
;	$H	書庫ﾄﾞﾗｲﾌﾞ			A:
;	$P	書庫ﾃﾞｨﾚｸﾄﾘ			\DIR\
;	$G	書庫ﾌｧｲﾙ名 (ﾌｧｲﾙ) 拡張子なし	ARC
;	$A	書庫ﾌｧｲﾙ名 (ﾄﾞﾗｲﾌﾞ/ﾌｧｲﾙ)	A:ARC.PMA
;	$E	書庫ﾌｧｲﾙ名 (ﾄﾞﾗｲﾌﾞ/ﾊﾟｽ/ﾌｧｲﾙ)	A:\DIR\ARC.PMA
;	$B	解凍ﾌｧｲﾙ名 (ﾄﾞﾗｲﾌﾞ/ﾌｧｲﾙ)	A:FILENAME.EXT
;	$C	解凍ﾌｧｲﾙ名 (ﾌｧｲﾙ)		FILENAME.EXT
;	$S	解凍ﾌｧｲﾙ名 (ﾌｧｲﾙ) 拡張子なし	FILENAME
;	$F	解凍ﾄﾞﾗｲﾌﾞ			A:
;	$I	解凍ﾄﾞﾗｲﾌﾞ/ﾃﾞｨﾚｸﾄﾘ
;	$D	ﾃﾝﾎﾟﾗﾘﾄﾞﾗｲﾌﾞ
;	$O	ｶﾚﾝﾄﾄﾞﾗｲﾌﾞ
;	$N	ｶﾚﾝﾄﾄﾞﾗｲﾌﾞ/ﾃﾞｨﾚｸﾄﾘ
;	$K	実行後、ｷｰ入力待ち
;	$R	追加ﾊﾟﾗﾒｰﾀを入力
;	$$	'$'1文字を表わす

;--------------
; 各種ツール類 ---------------------------------------------------------------
;-----------------| 32文字以内                   |----------------------------
editor		= KID $B			  ; ｴﾃﾞｨﾀ
pager		= VF $B				  ; ﾌｧｲﾙ表示ﾂｰﾙ
pma_extract	= PMEXT2 $A $B/I/M/N/W		  ; .PMA の解凍ﾂｰﾙ
pma_pack	= PMARC2 $A/$D $B		  ; .PMA の圧縮ﾂｰﾙ
pma_delete	= PMARC2 $A/$D $B/D		  ; .PMA の削除ﾂｰﾙ
lzh_extract	= PMEXT2 $A $B/I/M/N/W		  ; .LZH の解凍ﾂｰﾙ
lzh_pack	= LHPACK $A $B			  ; .LZH の圧縮ﾂｰﾙ
lzh_delete	= PMARC2 $A/$D $B/D		  ; .LZH の削除ﾂｰﾙ
lh5_pack	= PMARC2 $A/$D $B		  ; -lh5-の圧縮ﾂｰﾙ
lh5_extract	= PMEXT $A $B/I/M/N/W		  ; -lh5-の解凍ﾂｰﾙ
lzh_crccheck	= PMEXT $A $B/C/I/N$K		  ; .LZH のCRCﾁｪｯｸ
pma_crccheck	= PMEXT $A $B/C/I/N$K		  ; .PMA のCRCﾁｪｯｸ
make_sfx	= PMCAT $G.COM=PMSFX2.COM $A	  ; 自己展開書庫作成
ankfontfile	= 				  ; ANKﾌｫﾝﾄﾌｧｲﾙ

;-----------------------------------------------------------------------------
red		= 1	; 背景の赤の階調 (0〜7)
green		= 1	;   〃  緑の階調 (0〜7)
blue		= 3	;   〃  青の階調 (0〜7)
markchr		= 	; ﾏｰｸ時の表示ｷｬﾗｸﾀ (省略時は矢印)

;------------------------------------------------- 0(False)-- 1(True)---------
short_msg	= 0	; 複数解凍時の表示形式	   通常       簡略
lower		= 0	; ﾌｧｲﾙ名小文字表示	   しない     する
interlace_mode	= 0	; ｲﾝﾀｰﾚｰｽ		   しない     する
key_click	= 0	; ｷｰｸﾘｯｸ音		   しない     する
tree_fdisp	= 1	; TREEでのﾌｧｲﾙ表示         OFF        ON
nouse_mapper	= 0	; 子ﾌﾟﾛｾｽ時のﾏｯﾊﾟRAM(16KB) 使用       未使用
nouse_viewfont	= 0	; MSXView用漢字ROM	   使用       未使用
use_mouse	= 1	; ﾏｳｽ使用		   未使用     使用
language	= 1	; 使用言語(Language)	   Japanese   English
30		= 0	; 画面拡張モード	   通常(25行) 拡張(30行)

;-----------------|12文字以内|------------------------------------------------
hisfile		= PMM.HIS		; ｺﾏﾝﾄﾞ･ﾋｽﾄﾘｰ
phsfile		= PMM.PHS		; ﾊﾟｽ･ﾋｽﾄﾘｰ
treefile	= MMTREE.DIR		; TREEﾃﾞｰﾀ保存用

;----------------
; 拡張子対応情報 -------------------------------------------------------------
;---------|      48文字以内                              |--------------------
.COM	= $B$R
.LZH	= $O^M^JCD $N^M^JPMM $I$C^M^J$F^M^JCD $I
.PMA	= $O^M^JCD $N^M^JPMM $I$C^M^J$F^M^JCD $I
.MGS	= MGSQC $B /P
.SC?	= BLS $B
.MAG	= MAG $B
.PIC	= SLV $B
.Q4	= XLD4 $B
.JPG	= JLD $B
