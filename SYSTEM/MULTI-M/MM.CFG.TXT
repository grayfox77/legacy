;=========================================================================
; Pour faire la configuration de MM, veuillez utiliser CS.COM ver3.11
; de M.IIta. Puis, mettez les fichiers, MM.COM (non-compresse),
; MM.EXE , MM.CFG et MMRET.DAT dans le meme repertoire.
; Pendant l'utilisation de MM, vous pouvez faire la configuration de MM
; en poussant [GRAPH] puis [0] (avec la configuration initiale). Bien sur,
; a condition que les voix d'acces aux repertoires (Path) soient correcte
;=========================================================================


;# Parametres pour la configuration #

;Mode d'affichage des fichiers au demarrage:

MODE	= 5		; 1=Mode1 / 2=Mode2 / 3=Mode3 / 5=Mode5 / 9=Automatique

;Couleur des fichiers au demarrage:

MODE2	= 1		; 0 =Varie selon les attributs/ 1 =selon l'extention

;Attributs de l'affichage au premier demarrage:

ATTR_H  = 1		; Affichage des fichiers d'attribus invisibles
ATTR_S  = 1		; Affichage des fichiers d'attribus du system
ATTR_V  = 1		; Affichage du mom du volume
			; 0 =OFF / 1 =ON

;Mot de passe a l'execution (WILDCARD):

;	 0----5---10--+
WILDCARD= *.*         	; utilisation de "?","/","*" est possible

;Entree du son "clic":

KEYCLIK = 0		; 0 =OFF / 1 =ON

;Jeu de caracteres utilise pour l'affichage des Kanjis:

KANJIF	= 0		; 0 =Caracteres interne / 1 = ROM MSX-View

;Utilisation d'un sauveur d'ecran (screen saver):

;	 0----5---10--+
SCSAVER = MANBOW 	; Nom du sauveur d'ecran a executer. Si il n'y a
			; rien, celui interne a MM sera execute.
SC_TIME = 15		; temps d'attente avant l'execution 0= OFF / 1= 4.2 sec

;Affichage des fichiers d'un repertoire lors de l'arbre (TREE):

TREEFILE= 1		; 0 =OFF / 1 =ON

;Triage des fichiers avant l'affichage:

SORT    = 0	; 0 = pas de triage
		; 1 = triage en fonction du nom des fichiers
		; 2 = triage en fonction de la taille des fichiers
		; 3 = triage en fonction de la date de l'enregistrement
		; 4 = triage en fonction de l'extention du nom

;Indication des lecteurs utilises lors d'un arbre:

ATREE   = 1		; Lecture au lecteur	A
BTREE   = 1		;         "		B
CTREE   = 1		;         "		C
DTREE   = 1		;         "		D
ETREE   = 1		;         "		E
FTREE   = 1		;         "		F
GTREE   = 1		;         "		G
HTREE   = 1		;         "		H
;	0 = OFF / 1 = ON

;Utilisation du "Memory mapper" primaire libre:

;USERAM = 1		;0 =OFF / 1 =ON (Inutilise pour cette version)
 
;Mode d'affichage lors de la visualisation d'un texte:

VIEW_B  = 0		; Mode binaire
VIEW_T  = 1		; Longueur de la tabulation ( 1 = 4 caracteres)
VIEW_C  = 0		; Indication des codes [CTRL], interdit le mode ESC
VIEW_R  = 0		; Indication du code [RETURN]
VIEW_A  = 0		; Mode ANK
;	0 =OFF / 1 =ON


;# Creation d'un outil exterieur #


;Decompression de fichier:

;	 0----5---10---15---20---25---30--+
UNPMA   = PMEXT2 $Z $A$J$K				; Fichier .PMA
UNLHA   = PMEXT2 $Z $A$J$K				; Fichier .LZH
UNARC   = UNARC $Z $A$J$K			; Fichier .ARC / .ARK
UNISH   = ISH $Z /c$A /*$A$K			; Fichier .ISH
UNZIP   = UNZIP $Z $A$J$K			; Fichier .ZIP
RESTCOM = POPCOM -R $Z$K			; Fichier light .COM (DIET)


;Compression de fichier:

;	 0----5---10---15---20---25---30--+
PMA     = PMARC2 $F/$D $M$K			; .PMA
ISH     = ISH $T /ss /c $A$K			; Creation de fichier .ISH
LHA     = LHPACK $F $M$K			; Fichier .LZH
COMPCOM = POPCOM -D $T$K			; Fichier light .COM (DIET)


;Autre:

;	 0----5---10---15---20---25---30--+
CHKPMA  = PMM $T				; Decompression  [SHIFT]+[U]
EDITOR	= AKID $C				; Editeur	 ([SHIFT]+)[E]
PAGER   = $V					; Visu. texte    [V]
PAGER2  = NV $C					; Visu. texte    [SHIFT]+[V]
S_TOOL  = EDIT $C				; Outil No.1     [SHIFT]+[RET]
C_TOOL  = EDIT $C				; Outil No.2     [CTRL]+[RET]
SC_TOOL = CS $X					; Outil No.3  [SHI]+[CTR]+[RET]
TOOL    = $V					; Execution d'un outil en
;						  fonction de l'extention

;Liste des commandes du MACRO utilisable:

;	$A =	Nom de fichier de sortie (efficace seulement si on a entre un
;       nom de fichier)
;	$F =	Nom du fichier que l'on n'a entre (efficace seulement si on
;	a entre un nom de fichier)
;	$C =	Nom du fichier au curseur (avec son extention)
;	$X =	Nom du fichier au curseur (sans son extention)
;	$D =	Nom du lecteur present (curent drive)
;	$d =	Nom du lecteur temporaire
;	$J =	Execution en mode kanji
;	$K =	Attend l'appui d'une touche avant le retour a MM
;	$k =	Attend l'appui d'une touche apres chaque commandes
;	$P =	Voix (path) du repertoire present
;	$R =	Mise a la ligne d'une serie
;	$T =	Realisation d'une serie de fichiers labels (avec extention)
;	$t =	Realisation d'une serie de fichiers labels (sans extention)
;	$M =	Rassemble en une ligne et execute les fichiers labels (avec
;		leurs extentions)
;	$m =	Rassemble en une ligne et execute les fichiers labels (sans
;		leurs extentions)
;	$V =	Voir le texte du fichier present (il faut mettre en tete de
;		ligne)
;	$$ =	Entree de "$"
;	$Z =	Nom des fichiers labels (s'utilise uniquement lors de la
;		decompression d'une archive)
